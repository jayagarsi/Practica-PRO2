174 double Cjt_especies::calcular_distancia(const string& id1, const string& id2) {
175 
176  map<string, Especie>::iterator it;
177 
178  it = mesp.find(id1);
179  map<string, int> kmer1;
180  it->second.consultar_kmer(kmer1);
181 
182  it = mesp.find(id2);
183  map<string, int> kmer2;
184  it->second.consultar_kmer(kmer2);
185 
186  double cardinal_unio = 0;
187  double cardinal_interseccio = 0;
188 
189  map<string, int>::iterator it1 = kmer1.begin();
190  map<string, int>::iterator it2 = kmer2.begin();
191 
192  while (it1 != kmer1.end() and it2 != kmer2.end()) { // Calcul del cardinal de la unio i de la interseccio
193  // dels dos conjunts de kmers fent un recorregut
194  if (it1 -> first < it2 -> first) { // simultani als dos mapes
195  cardinal_unio += it1 -> second;
196  ++it1;
197  }
198 
199  else if (it2 -> first < it1 -> first) {
200  cardinal_unio += it2 -> second;
201  ++it2;
202  }
203 
204  else {
205  cardinal_unio += max(it1 ->second, it2 -> second);
206  cardinal_interseccio += min(it1 -> second, it2 ->second);
207  ++it1;
208  ++it2;
209  }
210  }
211 
212  while (it1 != kmer1.end()) { // En el cas que hi hagin mes elemetns en un conjunt que en l'altre
213  cardinal_unio += it1 -> second; // hem de sumar al cardinal de la unio aquells elements que
214  ++it1; // te de mes un dels dos conjunts
215  }
216 
217  while (it2 != kmer2.end()) {
218  cardinal_unio += it2 -> second;
219  ++it2;
220  }
221 
222  double distancia = (1 - (cardinal_interseccio/cardinal_unio))*100; // Formula de la distancia entre especies
